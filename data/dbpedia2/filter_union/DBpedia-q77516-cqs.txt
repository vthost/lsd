BASE    <file:///Users/Veronika/Desktop/git/qed/qed-core/>
PREFIX  dbont: <http://dbpedia.org/ontology/>
PREFIX  dbp:  <http://dbpedia.org/property/>
PREFIX  rdf:  <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>

CONSTRUCT 
  { 
    ?link rdf:type dbont:MusicalArtist .
    ?link rdf:type dbont:Band .
    ?link dbp:name ?name .
    ?link dbont:artist ?alba .
  }
WHERE
  { ?link  dbont:artist  ?alba
    FILTER regex(?name, "beatles", "i")
    ?link  dbp:name  ?name
    { FILTER NOT EXISTS { ?link  rdf:type  dbont:MusicalArtist }
      FILTER NOT EXISTS { ?link  rdf:type  dbont:Band }
    }
  }
LIMIT   3
