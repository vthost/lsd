CONSTRUCT 
  { 
    :THE_ELEPHANT_MAN dbpedia-prop:redirect ?movie .
    ?movie rdfs:label "THE ELEPHANT MAN"@en .
    ?movie dbpedia-prop:name "THE ELEPHANT MAN"@en .
    ?movie foaf:name "THE ELEPHANT MAN"@en .
    ?movie rdfs:label "THE ELEPHANT MAN (film)"@en .
    ?movie rdfs:label "The Elephant Man"@en .
    ?movie rdfs:label "The Elephant Man (film)"@en .
    ?movie foaf:name "The Elephant Man"@en .
    ?movie rdf:type dbpedia-owl:Film .
    ?movie foaf:page ?linkUrl .
    ?movie owl:sameAs ?sameAsUri .
    ?movie dbpedia-owl:imdbid ?imdbId .
    ?movie dbpedia-owl:releaseDate ?releaseDate .
  }
WHERE
  { { ?movie  dbpedia-owl:releaseDate  ?releaseDate }
    { ?movie  dbpedia-owl:imdbid  ?imdbId }
    { { FILTER NOT EXISTS { ?movie  foaf:page  ?linkUrl }
        FILTER NOT EXISTS { ?movie  owl:sameAs  ?sameAsUri }
      }
    }
    { ?movie  rdf:type  dbpedia-owl:Film }
    { FILTER NOT EXISTS { :THE_ELEPHANT_MAN
                                    dbpedia-prop:redirect  ?movie
                        }
      FILTER NOT EXISTS { ?movie  rdfs:label  "THE ELEPHANT MAN"@en }
      FILTER NOT EXISTS { ?movie  dbpedia-prop:name  "THE ELEPHANT MAN"@en }
      FILTER NOT EXISTS { ?movie  foaf:name  "THE ELEPHANT MAN"@en }
      FILTER NOT EXISTS { ?movie  rdfs:label  "THE ELEPHANT MAN (film)"@en }
      FILTER NOT EXISTS { ?movie  rdfs:label  "The Elephant Man"@en }
      FILTER NOT EXISTS { ?movie  rdfs:label  "The Elephant Man (film)"@en }
