CONSTRUCT 
  { 
    :Wimbledon dbpedia-prop:redirect ?event .
    ?event rdfs:label "Wimbledon"@en .
    ?event dbpedia-prop:name "Wimbledon"@en .
    ?event foaf:name "Wimbledon" .
    ?event rdfs:label "Wimbledon"@en .
    ?event rdfs:label "The Wimbledon"@en .
    ?event foaf:name "The Wimbledon" .
    ?event foaf:name "Wimbledon" .
    ?event rdf:type dbpedia-owl:SportsEvent .
    ?event rdf:type ?eventType .
    ?event skos:subject ?subject .
    ?event dbpedia-prop:wikiPageUsesTemplate ?template .
    ?event foaf:page ?linkUrl .
    ?event owl:sameAs ?sameAsUri .
    ?event dbpedia-prop:founded ?founded .
  }
WHERE
  { { ?event  dbpedia-prop:founded  ?founded }
    { { FILTER NOT EXISTS { ?event  foaf:page  ?linkUrl }
        FILTER NOT EXISTS { ?event  owl:sameAs  ?sameAsUri }
      }
    }
    { FILTER NOT EXISTS { ?event  rdf:type  dbpedia-owl:SportsEvent }
      FILTER NOT EXISTS { ?event  rdf:type  ?eventType
                          FILTER regex(?eventType, "http://dbpedia.org/class/yago/.*((GolfTours)|(Tournament)|(PGATour)|(SportsFestival)|(Competition)|(Championship)|(SportsEvent)|(Racing)|(Marathon)).*", "i")
                        }
      FILTER NOT EXISTS { ?event  skos:subject  ?subject
                          FILTER regex(?subject, "http://dbpedia.org/resource/Category:.*((games)|(golf_tours)|(grand_prix)|(tournaments)|(pga_tour)|(competitions)|(sporting_events)|(championship)|(racing)|(marathon)).*", "i")
                        }
      FILTER NOT EXISTS { ?event  dbpedia-prop:wikiPageUsesTemplate  ?template
                          FILTER regex(?template, "http://dbpedia.org/resource/Template:.*((tournament)|(competition)|(race)).*", "i")
                        }
    }
    { FILTER NOT EXISTS { :Wimbledon  dbpedia-prop:redirect  ?event }
      FILTER NOT EXISTS { ?event  rdfs:label  "Wimbledon"@en }
      FILTER NOT EXISTS { ?event  dbpedia-prop:name  "Wimbledon"@en }
      FILTER NOT EXISTS { ?event  foaf:name  "Wimbledon" }
      FILTER NOT EXISTS { ?event  rdfs:label  "Wimbledon"@en }
      FILTER NOT EXISTS { ?event  rdfs:label  "The Wimbledon"@en }
      FILTER NOT EXISTS { ?event  foaf:name  "The Wimbledon" }
