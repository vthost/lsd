  { 
    :Champions_League dbpedia-prop:redirect ?league .
    ?league rdfs:label "Champions League"@en .
    ?league dbpedia-prop:name "Champions League"@en .
    ?league foaf:name "Champions League" .
    ?league rdfs:label "Champions League"@en .
    ?league foaf:name "Champions League" .
    ?league rdf:type ?leagueType .
    ?league skos:subject ?subject .
    ?league dbpedia-prop:wikiPageUsesTemplate ?template .
    ?league foaf:page ?linkUrl .
    ?league owl:sameAs ?sameAsUri .
    ?league dbpedia-prop:founded ?founded .
    ?league dbpedia-prop:teams ?teams .
  }
WHERE
  { { FILTER ( ( datatype(?teams) = xsd:integer ) || ( datatype(?teams) = xsd:string ) )
      ?league  dbpedia-prop:teams  ?teams

    }
    { { FILTER NOT EXISTS { ?league  foaf:page  ?linkUrl }
        FILTER NOT EXISTS { ?league  owl:sameAs  ?sameAsUri }
      }
    }
    { FILTER NOT EXISTS { ?league  rdf:type  ?leagueType
                          FILTER regex(?leagueType, "http://dbpedia.org/class/yago/.*((SportsLeague)|(ballLeague)).*", "i")
                        }
      FILTER NOT EXISTS { ?league  skos:subject  ?subject
                          FILTER regex(?subject, "http://dbpedia.org/resource/Category:.*((Sports_leagues)|(ball_League)).*", "i")
                        }
      FILTER NOT EXISTS { ?league  dbpedia-prop:wikiPageUsesTemplate  ?template
                          FILTER regex(?template, "http://dbpedia.org/resource/Template:.*((sports_league)|(football_league)).*", "i")
                        }
    }
    { FILTER NOT EXISTS { :Champions_League
                                    dbpedia-prop:redirect  ?league
                        }
      FILTER NOT EXISTS { ?league  rdfs:label  "Champions League"@en }
      FILTER NOT EXISTS { ?league  dbpedia-prop:name  "Champions League"@en }
      FILTER NOT EXISTS { ?league  foaf:name  "Champions League" }
