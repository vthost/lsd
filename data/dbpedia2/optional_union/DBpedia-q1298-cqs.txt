CONSTRUCT 
  { 
    ?place rdfs:label "Merced County"@en .
    ?place rdfs:label "Merced County, California"@en .
    ?place dbpedia-prop:officialName "Merced County"@en .
    ?place dbpedia-prop:name "Merced County"@en .
    ?place foaf:name "Merced County"@en .
    ?place owl:sameAs <http://rdf.freebase.com/ns/guid.9202a8c04000641f8000000000090a27> .
    ?place rdf:type ?type .
    ?place dbpedia-prop:wikiPageUsesTemplate <http://dbpedia.org/resource/Template:coord> .
    ?place rdfs:label ?label .
    ?place geo:lat ?latitude .
    ?place geo:long ?longitude .
    ?place owl:sameAs ?sameAsUri .
    ?place foaf:page ?linkUrl .
  }
WHERE
  { { { FILTER NOT EXISTS { ?place  geo:lat   ?latitude ;
                                    geo:long  ?longitude
                          }
        FILTER NOT EXISTS { ?place  owl:sameAs  ?sameAsUri }
        FILTER NOT EXISTS { ?place  foaf:page  ?linkUrl }
      }
    }
    { FILTER ( lang(?label) = "en" )
      ?place  rdfs:label  ?label
    }
    { { FILTER NOT EXISTS { ?place  rdf:type  ?type
                            FILTER regex(?type, ".+/((Place)|(PopulatedPlace))$", "i")
                          }
        FILTER NOT EXISTS { ?place  dbpedia-prop:wikiPageUsesTemplate  <http://dbpedia.org/resource/Template:coord> }
      }
    }
    { { FILTER NOT EXISTS { ?place  rdfs:label  "Merced County"@en }
        FILTER NOT EXISTS { ?place  rdfs:label  "Merced County, California"@en }
        FILTER NOT EXISTS { ?place  dbpedia-prop:officialName  "Merced County"@en }
        FILTER NOT EXISTS { ?place  dbpedia-prop:name  "Merced County"@en }
        FILTER NOT EXISTS { ?place  foaf:name  "Merced County"@en }
        FILTER NOT EXISTS { ?place  owl:sameAs  <http://rdf.freebase.com/ns/guid.9202a8c04000641f8000000000090a27> }
      }
    }

PREFIX  dbpedia-prop: <http://dbpedia.org/property/>
PREFIX  geo:  <http://www.w3.org/2003/01/geo/wgs84_pos#>
PREFIX  rdf:  <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>
PREFIX  dc:   <http://purl.org/dc/elements/1.1/>

CONSTRUCT 
  { 
    ?place rdfs:label "Merced County"@en .
    ?place rdfs:label "Merced County, California"@en .
    ?place dbpedia-prop:officialName "Merced County"@en .
    ?place dbpedia-prop:name "Merced County"@en .
    ?place foaf:name "Merced County"@en .
    ?place owl:sameAs <http://rdf.freebase.com/ns/guid.9202a8c04000641f8000000000090a27> .
    ?place rdf:type ?type .
    ?place dbpedia-prop:wikiPageUsesTemplate <http://dbpedia.org/resource/Template:coord> .
    ?place rdfs:label ?label .
    ?place geo:lat ?latitude .
    ?place geo:long ?longitude .
    ?place owl:sameAs ?sameAsUri .
    ?place foaf:page ?linkUrl .
  }
WHERE
  { FILTER NOT EXISTS {   { ?place  geo:lat   ?latitude ;
                                    geo:long  ?longitude
                          }
                        UNION
                          { ?place  owl:sameAs  ?sameAsUri }
                        UNION
                          { ?place  foaf:page  ?linkUrl }
                      }
    { FILTER ( lang(?label) = "en" )
      ?place  rdfs:label  ?label
    }
    { { FILTER NOT EXISTS { ?place  rdf:type  ?type
                            FILTER regex(?type, ".+/((Place)|(PopulatedPlace))$", "i")
                          }
        FILTER NOT EXISTS { ?place  dbpedia-prop:wikiPageUsesTemplate  <http://dbpedia.org/resource/Template:coord> }
      }
    }
    { { FILTER NOT EXISTS { ?place  rdfs:label  "Merced County"@en }
        FILTER NOT EXISTS { ?place  rdfs:label  "Merced County, California"@en }
        FILTER NOT EXISTS { ?place  dbpedia-prop:officialName  "Merced County"@en }
        FILTER NOT EXISTS { ?place  dbpedia-prop:name  "Merced County"@en }
        FILTER NOT EXISTS { ?place  foaf:name  "Merced County"@en }
        FILTER NOT EXISTS { ?place  owl:sameAs  <http://rdf.freebase.com/ns/guid.9202a8c04000641f8000000000090a27> }
      }
    }
  }
