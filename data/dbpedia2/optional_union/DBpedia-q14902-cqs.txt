  { 
    ?medium rdfs:label "Better Homes and Gardens"@en .
    ?medium rdfs:label "Better Homes and Gardens (magazine)"@en .
    ?medium rdfs:label "Better Homes and Gardens (newspaper)"@en .
    ?medium foaf:name "Better Homes and Gardens" .
    ?medium rdfs:label "Better Homes and Gardens"@en .
    ?medium rdfs:label "The Better Homes and Gardens"@en .
    ?medium foaf:name "Better Homes and Gardens" .
    ?medium rdf:type dbpedia-owl:Newspaper .
    ?medium rdf:type dbpedia-owl:Magazine .
    ?medium rdf:type ?mediumType .
    ?medium skos:subject ?subject .
    ?medium dbpedia-prop:wikiPageUsesTemplate <http://dbpedia.org/resource/Template:infobox_newspaper> .
    ?medium dbpedia-prop:wikiPageUsesTemplate <http://dbpedia.org/resource/Template:infobox_magazine> .
    ?medium foaf:page ?linkUrl .
    ?medium owl:sameAs ?sameAsUri .
    ?medium dbpedia-prop:firstdate ?firstDate .
  }
WHERE
  { { ?medium  dbpedia-prop:firstdate  ?firstDate }
    { { FILTER NOT EXISTS { ?medium  foaf:page  ?linkUrl }
        FILTER NOT EXISTS { ?medium  owl:sameAs  ?sameAsUri }
      }
    }
    { FILTER NOT EXISTS { ?medium  rdf:type  dbpedia-owl:Newspaper }
      FILTER NOT EXISTS { ?medium  rdf:type  dbpedia-owl:Magazine }
      FILTER NOT EXISTS { ?medium  rdf:type  ?mediumType
                          FILTER regex(?mediumType, "((http://umbel.org/umbel/sc/)|(http://dbpedia.org/class/yago/)).*((Newspaper)|(Magazine)).*", "i")
                        }
      FILTER NOT EXISTS { ?medium  skos:subject  ?subject
                          FILTER regex(?subject, "http://dbpedia.org/resource/Category.*((newspapers)|(magazines)|(journals)).*", "i")
                        }
      FILTER NOT EXISTS { ?medium  dbpedia-prop:wikiPageUsesTemplate  <http://dbpedia.org/resource/Template:infobox_newspaper> }
      FILTER NOT EXISTS { ?medium  dbpedia-prop:wikiPageUsesTemplate  <http://dbpedia.org/resource/Template:infobox_magazine> }
    }
    { FILTER NOT EXISTS { ?medium  rdfs:label  "Better Homes and Gardens"@en }
      FILTER NOT EXISTS { ?medium  rdfs:label  "Better Homes and Gardens (magazine)"@en }
      FILTER NOT EXISTS { ?medium  rdfs:label  "Better Homes and Gardens (newspaper)"@en }
      FILTER NOT EXISTS { ?medium  foaf:name  "Better Homes and Gardens" }
      FILTER NOT EXISTS { ?medium  rdfs:label  "Better Homes and Gardens"@en }
      FILTER NOT EXISTS { ?medium  rdfs:label  "The Better Homes and Gardens"@en }
      FILTER NOT EXISTS { ?medium  foaf:name  "Better Homes and Gardens" }
