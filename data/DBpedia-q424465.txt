PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  owl:  <http://www.w3.org/2002/07/owl#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>

SELECT DISTINCT  ?title ?description ?sameas
WHERE
  { <http://dbpedia.org/resource/Aaron_Sorkin>
              foaf:name     ?title ;
              rdfs:comment  ?description
    FILTER ( lang(?description) = "en" )
    OPTIONAL
      {   { ?sameas  owl:sameAs  <http://dbpedia.org/resource/Aaron_Sorkin> }
        UNION
          { <http://dbpedia.org/resource/Aaron_Sorkin>
                      owl:sameAs  ?sameas
          }
      }
  }
