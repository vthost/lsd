  { 
    ?album p:artist <http://dbpedia.org/resource/Black_Sabbath> .
    ?album rdf:type <http://dbpedia.org/class/yago/Album106591815> .
    ?album p:cover ?cover .
    ?album p:name ?name .
    ?album p:released ?dateofrelease .
  }
WHERE
  { { ?album  p:released  ?dateofrelease }
    { ?album  p:name  ?name }
    { ?album  p:cover  ?cover }
    ?album  p:artist  <http://dbpedia.org/resource/Black_Sabbath> ;
            rdf:type  <http://dbpedia.org/class/yago/Album106591815>
  }
