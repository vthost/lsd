PREFIX  xsd:  <http://www.w3.org/2001/XMLSchema#>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  wd:   <http://www.wikidata.org/entity/>
PREFIX  wdt:  <http://www.wikidata.org/prop/direct/>

CONSTRUCT 
  { 
    ?subj wdt:P31 wd:Q571 .
    ?subj wdt:P31 wd:Q7725634 .
    ?subj wdt:P577 ?date .
    ?subj wdt:P291 ?place .
    ?subj wdt:P840 ?place .
    ?place wdt:P625 ?coord .
    ?subj rdfs:label ?label .
  }
WHERE
  { FILTER ( lang(?label) = "en" )
    ?subj  rdfs:label  ?label
    { ?place  wdt:P625  ?coord
      { FILTER NOT EXISTS { ?subj  wdt:P291  ?place }
        FILTER NOT EXISTS { ?subj  wdt:P840  ?place }
      }
    }
    FILTER ( ?date < "1830-01-01T00:00:00Z"^^xsd:dateTime )
    ?subj  wdt:P577  ?date
    { FILTER NOT EXISTS { ?subj  wdt:P31  wd:Q571 }
      FILTER NOT EXISTS { ?subj  wdt:P31  wd:Q7725634 }
    }
  }
LIMIT   3
