PREFIX  wikibase: <http://wikiba.se/ontology#>

  { 
    ?pl rdfs:label ?place .
    ?s rdfs:label ?writer .
    ?s wdt:P27 ?pl .
    ?s wikibase:sitelinks ?linkcount .

WHERE
  { { FILTER ( lang(?place) = "en" )
      ?pl  rdfs:label  ?place

    { FILTER ( lang(?writer) = "en" )
      ?s  rdfs:label  ?writer
    }
    ?s  wdt:P27             ?pl ;
        wikibase:sitelinks  ?linkcount
    { ?s  wdt:P106  wd:Q49757 }
  }
