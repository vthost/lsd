BASE    <file:///Users/Veronika/Desktop/git/qed/qed-core/>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  wd:   <http://www.wikidata.org/entity/>
PREFIX  wdt:  <http://www.wikidata.org/prop/direct/>

CONSTRUCT 
  { 
    ?h wdt:P39 wd:Q11696 .
    ?h wdt:P509 ?cause .
    ?h rdfs:label ?hl .
    ?cause rdfs:label ?causel .
  }
WHERE
  { { FILTER ( lang(?causel) = "en" )
      ?cause  rdfs:label  ?causel
    }
    { FILTER ( lang(?hl) = "en" )
      ?h  rdfs:label  ?hl
    }
    ?h  wdt:P39   wd:Q11696 ;
        wdt:P509  ?cause
  }
LIMIT   2
