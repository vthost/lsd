PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  wd:   <http://www.wikidata.org/entity/>
PREFIX  wdt:  <http://www.wikidata.org/prop/direct/>

CONSTRUCT 
  { 
    ?subj rdfs:label ?label .
    ?place50 wdt:P625 ?coord11 .
    ?subj9 wdt:P840 ?place50 .
    ?subj wdt:P577 ?date .
    ?subj wdt:P31 wd:Q571 .
    ?subj wdt:P31 wd:Q7725634 .
  }
WHERE
  { FILTER ( ! ( lang(?label) = "en" ) )
    ?subj  rdfs:label  ?label
    OPTIONAL
      { ?place50  wdt:P625  ?coord11
        { ?subj9  wdt:P840  ?place50 }
      }
    FILTER ( ! ( ?date < "1830-01-01T00:00:00Z"^^xsd:dateTime ) )
    ?subj  wdt:P577  ?date
    { { ?subj  wdt:P31  wd:Q571 }
      { ?subj  wdt:P31  wd:Q7725634 }
    }
    FILTER ( ( ( ( ?subj != ?subj9 ) || ( ?subj != ?subj9 ) ) || ( ?subj != ?subj9 ) ) || ( ?subj != ?subj9 ) )
  }
