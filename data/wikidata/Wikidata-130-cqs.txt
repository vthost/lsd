PREFIX  schema: <http://schema.org/>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  wd:   <http://www.wikidata.org/entity/>
PREFIX  wdt:  <http://www.wikidata.org/prop/direct/>

CONSTRUCT 
  { 
    ?s wdt:P106 wd:Q36180 .
    ?s wdt:P106 wd:Q482980 .
    ?s wdt:P106 wd:Q49757 .
    ?s wdt:P27 ?pl .
    ?sitelink schema:about ?s .
    ?s rdfs:label ?writer .
    ?pl rdfs:label ?place .
  }
WHERE
  { { FILTER ( lang(?place) = "en" )
      ?pl  rdfs:label  ?place
    }
    { FILTER ( lang(?writer) = "en" )
      ?s  rdfs:label  ?writer
    }
    ?s        wdt:P27       ?pl .
    ?sitelink  schema:about  ?s
    { FILTER NOT EXISTS { ?s  wdt:P106  wd:Q36180 }
      FILTER NOT EXISTS { ?s  wdt:P106  wd:Q482980 }
      FILTER NOT EXISTS { ?s  wdt:P106  wd:Q49757 }
    }
  }
LIMIT   2
