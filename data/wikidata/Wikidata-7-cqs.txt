PREFIX  wdt:  <http://www.wikidata.org/prop/direct/>

  { 
    ?subj rdfs:label ?label .
    ?subj wdt:P582 ?d3 .
    ?subj wdt:P585 ?d2 .
    ?subj wdt:P580 ?d1 .
    ?subj wdt:P31 wd:Q178561 .

  }

  { FILTER ( lang(?label) = "en" )

    BIND(if(( ! bound(?d1) ), if(( ! bound(?d2) ), ?d3, ?d2), ?d1) AS ?date)
    { ?subj  wdt:P582  ?d3 }

    { ?subj  wdt:P580  ?d1 }
    ?subj  wdt:P31   wd:Q178561 ;
           wdt:P625  ?coord
  }
