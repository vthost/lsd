BASE    <file:///Users/Veronika/Desktop/git/qed/qed-core/>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  defaultView: <http://defaultView>
PREFIX  wd:   <http://www.wikidata.org/entity/>
PREFIX  wdt:  <http://www.wikidata.org/prop/direct/>

CONSTRUCT 
  { 
    ?person wdt:P106 wd:Q188094 .
    ?person wd:P101 wd:Q8134 .
    ?person wdt:P106 wd:Q188094 .
    ?person wd:P101 wd:Q8134 .
    ?person wdt:P19 ?place .
    ?place wdt:P625 ?coord .
    ?person wdt:P569 ?dob .
    ?person rdfs:label ?name .
    ?place rdfs:label ?birthplace .
  }
WHERE
  { FILTER ( lang(?birthplace) = "en" )
    ?place  rdfs:label  ?birthplace
    FILTER ( lang(?name) = "en" )
    ?person  rdfs:label  ?name
    BIND(year(?dob) AS ?birthyear)
    { ?person  wdt:P569  ?dob }
    ?person  wdt:P19   ?place .
    ?place   wdt:P625  ?coord
    { ?person  wdt:P106  wd:Q188094 ;
               wd:P101   wd:Q8134
    }
    { FILTER NOT EXISTS { ?person  wdt:P106  wd:Q188094 }
      FILTER NOT EXISTS { ?person  wd:P101  wd:Q8134 }
    }
  }
LIMIT   2
