  { 
    ?item ?prop ?statement .
    ?statement prov:wasDerivedFrom ?derivedFrom .
    ?item wdt:P279 wd:Q7187 .
    ?item wdt:P703 wd:Q5 .
    ?item ?prop ?statement .
  }
WHERE
  { { ?item     ?prop                ?statement .
      ?statement  prov:wasDerivedFrom  ?derivedFrom
    }
    FILTER regex(str(?prop), "http://www.wikidata.org/prop/")
    FILTER ( ! regex(str(?prop), "http://www.wikidata.org/prop/direct/") )
    ?item  wdt:P279  wd:Q7187 ;
           wdt:P703  wd:Q5 ;
           ?prop     ?statement
  }
