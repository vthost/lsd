PREFIX  dbo:  <http://dbpedia.org/ontology/>
PREFIX  rdf:  <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX  resource: <http://dbpedia.org/resource/>
PREFIX  property: <http://dbpedia.org/property/>
PREFIX  skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>

CONSTRUCT 
  { 
    ?id rdf:type dbo:River .
    ?id rdf:type dbo:Stream .
    ?id property:country ?country .
  }
WHERE
  { FILTER ( ( str(?country) = "United States" ) || ( ?country = resource:United_States ) )
    ?id  property:country  ?country
    { FILTER NOT EXISTS { ?id  rdf:type  dbo:River }
      FILTER NOT EXISTS { ?id  rdf:type  dbo:Stream }
    }
  }
LIMIT   10
