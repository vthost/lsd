BASE    <file:///Users/Veronika/Desktop/git/qed/qed/>
PREFIX  :     <http://dbpedia.org/resource/>
PREFIX  owl:  <http://www.w3.org/2002/07/owl#>
PREFIX  rdf:  <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX  xsd:  <http://www.w3.org/2001/XMLSchema#>
PREFIX  skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  dbpedia: <http://dbpedia.org/>
PREFIX  dbpedia2: <http://dbpedia.org/property/>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>
PREFIX  dc:   <http://purl.org/dc/elements/1.1/>

CONSTRUCT 
  { 
    ?subject rdf:type <http://dbpedia.org/ontology/Film> .
    ?subject dbpedia2:starring :Tom_Cruise .
    ?subject rdfs:comment ?abstract .
    ?subject rdfs:label ?label .
    ?subject <http://dbpedia.org/ontology/releaseDate> ?released .
  }
WHERE
  { FILTER ( xsd:date(?released) < "2010-01-01"^^xsd:date )
    ?subject  <http://dbpedia.org/ontology/releaseDate>  ?released
    FILTER ( ( lang(?abstract) = "en" ) && ( lang(?label) = "en" ) )
    ?subject  rdf:type           <http://dbpedia.org/ontology/Film> ;
              dbpedia2:starring  :Tom_Cruise ;
              rdfs:comment       ?abstract ;
              rdfs:label         ?label
  }
LIMIT   10
