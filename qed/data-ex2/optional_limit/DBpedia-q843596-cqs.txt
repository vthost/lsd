  { 
    ?sub rdfs:subClassOf ?super .
    ?sub rdfs:label ?sublabel .
    ?sup rdfs:label ?suplabel .
  }

  { { ?sup  rdfs:label  ?suplabel }
    { ?sub  rdfs:label  ?sublabel }
    ?sub  rdfs:subClassOf  ?super
  }
LIMIT   10
