PREFIX  ont:  <http://dbpedia.org/ontology/>
PREFIX  owl:  <http://www.w3.org/2002/07/owl#>
PREFIX  rdf:  <http://www.w3.org/1999/02/22-rdfsyntax-ns#>
PREFIX  prop: <http://dbpedia.org/property/>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>
PREFIX  db:   <http://dbpedia.org/resource/>

CONSTRUCT 
  { 
    ?show prop:network ?network .
    ?show prop:network ?network .
    ?show prop:showName ?name .
    ?show prop:numSeasons ?seasons .
  }
WHERE
  { FILTER ( ?seasons > 4 )
    ?show  prop:showName    ?name ;
           prop:numSeasons  ?seasons
    { FILTER NOT EXISTS { ?show  prop:network  ?network
                          FILTER regex(?network, "HBO")
                        }
      FILTER NOT EXISTS { ?show  prop:network  ?network
                          FILTER regex(?network, "Showtime")
                        }
    }
  }
LIMIT   10
