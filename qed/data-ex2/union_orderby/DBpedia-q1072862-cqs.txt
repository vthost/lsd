PREFIX  :     <http://dbpedia.org/resource/>
PREFIX  owl:  <http://www.w3.org/2002/07/owl#>
PREFIX  rdf:  <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX  xsd:  <http://www.w3.org/2001/XMLSchema#>
PREFIX  skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  dbpedia: <http://dbpedia.org/>
PREFIX  dbpedia2: <http://dbpedia.org/property/>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>
PREFIX  dc:   <http://purl.org/dc/elements/1.1/>

CONSTRUCT 
  { 
    ?x skos:subject <http://dbpedia.org/resource/Category:Horror_video_games> .
    ?x skos:subject <http://dbpedia.org/resource/Category:First-person_shooters> .
    ?x skos:subject <http://dbpedia.org/resource/Category:Horror_video_games> .
    ?x skos:subject <http://dbpedia.org/resource/Category:Third-person_shooters> .
    ?x rdfs:label ?y .
  }
WHERE
  { FILTER ( lang(?y) = "en" )
    ?x  rdfs:label  ?y
    { FILTER NOT EXISTS { ?x  skos:subject  <http://dbpedia.org/resource/Category:Horror_video_games> ;
                              skos:subject  <http://dbpedia.org/resource/Category:First-person_shooters>
                        }
      FILTER NOT EXISTS { ?x  skos:subject  <http://dbpedia.org/resource/Category:Horror_video_games> ;
                              skos:subject  <http://dbpedia.org/resource/Category:Third-person_shooters>
                        }
    }
  }
