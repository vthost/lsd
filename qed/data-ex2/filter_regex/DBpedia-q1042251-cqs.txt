PREFIX  rdf:  <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>
PREFIX  dc:   <http://purl.org/dc/elements/1.1/>

CONSTRUCT 
  { 
    :AFN dbpedia-prop:redirect ?cur .
    ?cur dbpedia-prop:isoCode "AFN"@en .
    ?cur rdfs:label "AFN"@en .
    ?cur dbpedia-prop:name "AFN"@en .
    ?cur foaf:name "AFN" .
    ?cur rdf:type dbpedia-owl:Currency .
    ?cur rdf:type ?curType .
    ?cur dbpedia-prop:wikiPageUsesTemplate ?template .
    ?cur owl:sameAs ?sameAsUri .
    ?cur foaf:page ?linkUrl .
  }
WHERE
  { { { FILTER NOT EXISTS { ?cur  owl:sameAs  ?sameAsUri }
        FILTER NOT EXISTS { ?cur  foaf:page  ?linkUrl }
      }
    }
    { FILTER NOT EXISTS { ?cur  rdf:type  dbpedia-owl:Currency }
      FILTER NOT EXISTS { ?cur  rdf:type  ?curType
                          FILTER regex(?curType, "http://dbpedia.org/class/yago/Currency.*", "i")
                        }
      FILTER NOT EXISTS { ?cur  dbpedia-prop:wikiPageUsesTemplate  ?template
                          FILTER regex(?template, "http://dbpedia.org/resource/Template:infobox_currency", "i")
                        }
    }
    { FILTER NOT EXISTS { :AFN  dbpedia-prop:redirect  ?cur }
      FILTER NOT EXISTS { ?cur  dbpedia-prop:isoCode  "AFN"@en }
      FILTER NOT EXISTS { ?cur  rdfs:label  "AFN"@en }
      FILTER NOT EXISTS { ?cur  dbpedia-prop:name  "AFN"@en }
      FILTER NOT EXISTS { ?cur  foaf:name  "AFN" }
