BASE    <file:///Users/Veronika/Desktop/git/qed/qed/>
PREFIX  skos: <http://www.w3.org/2004/02/skos/core#>

CONSTRUCT 
  { 
    ?tip skos:subject <http://dbpedia.org/resource/Category:Computer_printers> .
    ?tip <http://dbpedia.org/ontology/abstract> ?abstract .
    ?tip <http://dbpedia.org/ontology/thumbnail> ?poza .
  }
WHERE
  { ?tip  <http://dbpedia.org/ontology/thumbnail>  ?poza
    FILTER regex(?abstract, "black")
    FILTER langMatches(lang(?abstract), "en")
    ?tip  skos:subject          <http://dbpedia.org/resource/Category:Computer_printers> ;
          <http://dbpedia.org/ontology/abstract>  ?abstract
  }
LIMIT   10
