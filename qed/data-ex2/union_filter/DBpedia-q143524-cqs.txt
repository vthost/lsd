BASE    <file:///Users/Veronika/Desktop/git/qed/qed/>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  owl:  <http://www.w3.org/2002/07/owl#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>

CONSTRUCT 
  { 
    <http://dbpedia.org/resource/Meet_Joe_Black> foaf:name ?title .
    <http://dbpedia.org/resource/Meet_Joe_Black> rdfs:comment ?description .
    ?sameas owl:sameAs <http://dbpedia.org/resource/Meet_Joe_Black> .
    <http://dbpedia.org/resource/Meet_Joe_Black> owl:sameAs ?sameas .
  }
WHERE
  { { { FILTER NOT EXISTS { ?sameas  owl:sameAs  <http://dbpedia.org/resource/Meet_Joe_Black> }
        FILTER NOT EXISTS { <http://dbpedia.org/resource/Meet_Joe_Black>
                                      owl:sameAs  ?sameas
                          }
      }
    }
    FILTER ( lang(?description) = "en" )
    <http://dbpedia.org/resource/Meet_Joe_Black>
              foaf:name     ?title ;
              rdfs:comment  ?description
  }
LIMIT   10
