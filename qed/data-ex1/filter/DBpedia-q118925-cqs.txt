PREFIX  address: <http://schemas.talis.com/2005/address/schema#>
PREFIX  owl:  <http://www.w3.org/2002/07/owl#>
PREFIX  xsd:  <http://www.w3.org/2001/XMLSchema#>
PREFIX  skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  terms: <http://purl.org/dc/terms/>
PREFIX  rdf:  <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX  emocom: <http://www.emotion-and-computing.de/resource/>
PREFIX  timeline: <http://purl.org/NET/c4dm/timeline.owl#>
PREFIX  bibo: <http://purl.org/ontology/bibo/>
PREFIX  event: <http://purl.org/NET/c4dm/event.owl#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>

CONSTRUCT 
  { 
    <http://dbpedia.org/resource/List_of_non-fiction_writers> skos:subject ?resource .
    ?resource rdfs:label ?label .
    ?resource foaf:page ?page .
  }
WHERE
  { FILTER ( lang(?label) = "en" )
    { ?resource  foaf:page  ?page }
    <http://dbpedia.org/resource/List_of_non-fiction_writers>
              skos:subject  ?resource .
    ?resource  rdfs:label   ?label
  }
LIMIT   10
