PREFIX  geo:  <http://www.w3.org/2003/01/geo/wgs84_pos#>
PREFIX  xsd:  <http://www.w3.org/2001/XMLSchema#>

CONSTRUCT 
  { 
    ?l geo:lat ?lat .
    ?l geo:long ?lng .
    ?l <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> ?t .
  }
WHERE
  { ?l  geo:lat               ?lat ;
        geo:long              ?lng ;
        a                     ?t
  }
LIMIT   10
